Javascript Algorithms for Searching and Sorting



Throughout my years as a programmer, it seems as if every time I leave the realm of simple DOM manipulation and form submission, I find myself in the position where I need to search or sort large amounts of data. However, with the limited power of JavaScript, and the expectations for webpages to be lightweight, I ran into some major UX problems. Therefore, over time I have come up with the most efficient algorithms possible for doing this, and have decided to save everyone else the bother by posting it all here.


Note: All of the following functions are designed to work with array. If you need to sort or search text I would recommend first using the split() function.

function sort(arr, from, to) { 

  if (from < to){
    
      var middle = partition(arr, from, to);
      sort(arr, from, middle - 1);
      sort(arr, middle+1, to);
    
  }

  return arr; 
         
}


function partition(arr, from, to)
{
	var pivot = arr[from];
	var left = from + 1;
	var right = to;

	do 
	{
		while(arr[left] <= pivot && left < right)
			left++;

		while(arr[right] > pivot)
			right--;

        if (left < right){
			
            var b = arr[right];
            arr[right] = arr[left];
            arr[left] = b;

        }

	} while(left < right);

  
    var b = arr[right];
    arr[right] = arr[from];
    arr[from] = b;

    return right;
}

function binary_search(arr, start, finish, key)
{
   if (start > finish)
      return -1;

   var mid = Math.floor((start + finish) / 2);

   if (arr[mid] == key)
      return mid;
   else if (arr[mid] < key)    
      return binary_search(arr, mid + 1, finish, key);
   else                      
      return binary_search(arr, start, mid - 1, key);
}
